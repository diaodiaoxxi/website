# 原型链继承

  • 本质：复制，重写对象的原型
  • 缺点：子类构造函数创建的对象，其原型__proto__公用同一个原型对象，会造成多个实例对原型中引用类型进行的篡改
  • eg：原型中有数组时，实例通过arr[3], arr.push()等操作 ，可以直接改变原型上数组的值

(function(){

  function superType() {
    this.colors = ["red", "blue", "green"];
  }
  
  function subType() {}
  subType.prototype = new superType()
  
  var instance1 = new subType()
  // 缺点体现：直接改变subType原型上colors的值，由于实例公用同一个原型，影响了instance2获取colors值
  instance1.colors[3] = 'black'
  console.log(instance1.colors)  // "red", "blue", "green", 'black'
  var instance2 = new subType()
  console.log(instance2.colors)  // "red", "blue", "green", 'black'

})()

# 构造函数继承

  • 通过父类构造函数来增强子类的实例；等同于复制父类的实例给子类
  • 缺点：
      1. 不能继承父类构造函数的原型
      1. 无法实现复用，每个子类都有父类实例函数的副本，影响性能，占内存

 (function(){

  function superType() {
    this.colors = ["red", "blue", "green"];
  }

  function subType() {
    superType.call(this)
    this.name = 'subType'
  }

  var instance1 = new subType()
  instance1.colors[3] = 'black'
  console.log(instance1.colors)  // "red", "blue", "green", 'black'
  var instance2 = new subType()
  console.log(instance2.colors)  // "red", "blue", "green", 

 })()

# 组合继承（原型链继承，构造函数继承两个方式的组合）

  • 通过原型链进行原型属性和方法的继承
  • 构造函数来实现实例属性的继承
  • 缺点：子类在创建对象时，其原型链中会存在两份相同的属性/方法

  (function(){

    function superType(name) {
      this.name = name
      this.colors = ["red", "blue", "green"];
    }

    SuperType.prototype.sayName = function(){
      alert(this.name);
    };

    function subType(name, age) {
      superType.call(this, name)
      this.age = age
    }

    subType.prototype = new superType()

    subType.prototype.constructor = subType

    subType.prototype.getAge = function() {
      console.log(this.age);
    }

    var instance1 = new SubType("Nicholas", 29);
    var instance2 = new SubType("Greg", 27);  
   
  })()

# 原型式继承

  • 本质上：Object.create(obj)
  • 利用一个空对象作为中介，将某个对象直接赋值给空对象构造函数的原型
  • 缺点：
    1. 原型链上的引用类型属性容易被篡改
    1. 无法传递参数

  (function(){
    // 本质上 Object.create(obj) 可替代下方object函数

    function object(obj) {
      function F(){}
      F.prototype = obj
      return new F()
    }

    var person = {
      name: "Nicholas",
      friends: ["Shelby", "Court", "Van"]
    };

    var anotherPerson1 = object(person)
    anotherPerson1.name = "Greg";
    anotherPerson1.friends.push('test1')

    var anotherPerson2 = object(person)
    anotherPerson2.name = "Linda";
    anotherPerson2.friends.push('test2')

  })()

# 寄生式继承

  • 在原型式继承的基础上，增强子类创建对象的实例，返回构造函数
  • 缺点：
      1. 原型链上的引用类型属性容易被篡改
      1. 无法传递参数

   (function(){

    function object(obj) {
      function F(){}
      F.prototype = obj
      return new F()
    }

    function createAnother(origin) {
      var clone = object(origin)
      clone.sayHi = function(){  // 以某种方式来增强对象
        alert("hi");
      };
      return clone
    }

    var person = {
      name: "Nicholas",
      friends: ["Shelby", "Court", "Van"]
    };

    var anotherPerson = createAnother(person);
    anotherPerson.sayHi();
  
  })()

# 寄生组合式继承


   (function(){
    // 手写Object.create
    function createObject(obj) {
      function F(){}
      F.prototype = obj
      return new F()
    }

    function inheritProperty(subType, superType) {
      // 创建对象，创建父类原型的一个副本
      // 使用了 原型式的继承方式
      var property = Object.create(superType.prototype)
      // var property = createObject(superType.prototype)
      // 增强对象，补充因重写原型而失去的默认的constructor属性
      property.constructor = subType
      // 指定对象，将新创建的对象赋值给子类的原型
      // 使用了原型链的继承 方式
      subType.prototype = property

    }

    // 父类初始化实例属性和原型属性
    function superType(name) {
      this.name = name
      this.colors = ["red", "blue", "green"];
    }

    superType.prototype.getName = function() {
      return this.name
    }
    
    // 借用构造函数传递增强子类实例属性（支持传参和避免篡改）
    function subType(name,  age) {
      // 使用了构造函数的继承方式
      superType.call(this, name)
      this.age = age
    }

    inheritProperty(subType, superType)

    // 新增子类原型属性
    subType.prototype.sayAge = function(){
      alert(this.age);
    }

    var instance1 = new subType("xyc", 23);
    var instance2 = new subType("lxy", 23);
  
  })()

# 混入方式继承多个对象


   (function(){

    function MyClass() {
      SuperClass.call(this);
      OtherSuperClass.call(this);
    }
    // 继承一个类
    MyClass.prototype = Object.create(SuperClass.prototype)

    // MyClass.prototype本身对象的原型: 混入OtherSuperClass类
    Object.assign(MyClass.prototype, OtherSuperClass.prototype)

    // 重新指定constructor
    MyClass.prototype.constructor = MyClass

    MyClass.prototype.myMethod = function() {
      // do something
    }
  
  })()

# class extends继承


   (function(){

    class superType {
      constructor(width, height) {
        this.width = width
        this.height = height
        this.colors = ["red", "blue", "green"]
        this.name = 'superType'
      }
      get area() {
        return this.getArea()
      }
      getArea() {
        return this.width * this.height
      }
    }

    class subType extends superType {
      constructor(length) {
        super(length, length)
        this.name = 'subType'
      }
      get area() {
        return this.width * this.height
      }
    }

    const subtype = subType(10)
  
  })()
