# call && applay && bind


fn.call(obj, 1,2)
fn.apply(obj, [1,2])
fn.bind(obj)(1,2)

Function.prototype.myCall = function(context) {
  context = context || window
  // 确定this指向 转化为 obj.fn = fn
  context.fn = this
  const args = [...arguments].slice(1)
  // const args = Array.prototype.slice.call(arguments, 1)
  // fn.call(obj, 1,2) 转化为 obj.fn(1,2)
  const result = context.fn(...args)
  // 删除扩展的fn
  delete context.fn
  return result
}


Function.prototype.myApply = function(context) {
  context = context || window
  context.fn = this
  let result
  // 需要判断是否存储第二个参数
  // 如果存在，就将第二个参数展开
  if (arguments[1]) {
    result = context.fn(...arguments[1])
  } else {
    result = context.fn()
  }
   // 删除扩展的fn
  delete context.fn
  return result
}


Function.prototype.myBind = function(context) {
  if (typeof this !== 'function') {
    throw new Error('Error')
  }
  const _this = this
  const args = [...arguments].slice(1)
  // 返回⼀个函数
  return function F() {
    // 因为返回了⼀个函数，我们可以 new F()，所以需要判断; 
    // 当使用new创建对象，bind绑定this失效
    if (this instanceof F) {
      return new _this(...args, ...arguments)
    }
    return _this.apply(context, args.concat(arguments))
  }
}

# 防抖、节流


/**
 * 高频事件触发接受收，n秒内触发一次函数
 * 场景：输入框输入文字，滚动事件绑定，窗口 resize，
 * @param {*} fn 
 * @param {*} timing 
 * @returns 
 */
function debounce(fn, timing) {
  let timer
  return function() {
    clearTimeout(timer)
    timer = setTimeout(() => {
      fn()
    }, timing);
  }
}

/**
 * 高频事件触发，但在 n 秒内只会执行一次，所以节流会稀释函数的执行效率。
 * 场景：滚动加载更多，搜索框的联想功能、高频点击、表单重复提交
 * @param {*} fn 
 * @param {*} timing 
 * @returns 
 */
function throttle(fn, timing){
  let trigger
  return function() {
    if (trigger) return
    trigger = true
    fn()
    setTimeout(() => {
      trigger = false
    }, timing);
  }
}

# 深拷贝


function deepCopy(source = {}) {
  // 对象不是obj对象的 return typepf 为 object 有 object、array 区别null
  //  obj 是 null ，或者不是对象和数组，直接返回
  if (source === null || typeof source !== 'object') {
    return source
  }
  // var newSource = source.constructor === Array ? [] : {};
  var newSource = source instanceof Array ? [] : {};
  for (let key in source) {
    // 判断key是否为自身属性; 过滤原型上的属性
    if (source.hasOwnProperty(key)) {
      if (typeof source[key] === 'object'){
        newSource[key] = deepCopy(source[key])
      } else {
        newSource[key] = source[key]
      }
    }
  }
  return newSource
}

  while(target){
    if (target.__proto__ === origin.prototype) {
      return true
    }
    target = target.__proto__ 
  }
  return false
}

function instanceOf(left, right) {
  // 传入的left目标对象是 null、以及是原始类型 则直接返回 false
  if (left === null ||( typeof left !== 'object' && typeof left !== 'function')){
    return false
  }
  right = right.prototype // 获取右边构造函数原型
  left = left.__proto__  //  获取左边实例对象的原型
  while(true){
    if (left === null) return false  // 若遍历原型链到顶部，还是没匹配到原型，则返回false
    if (left === right) return true // 若当前left的原型上找到当前的原型，则返回true
    left = left.__proto__  // 递归遍历，重新在上一级原型上进行匹配
  }
}

# new


function New(fn, ...args) {
  // 创建obj，且继承原型
  let obj = Object.create(fn.prototype)
  // 创建实例，绑定this
  var res = fn.apply(obj, args)
  // 确保new出来的是object对象
  return res instanceof Object ? res  : obj
}

/**
 * 创建一个对象 obj
 * 构造函数的原型赋值给obj对象实例的原型 obj.__proto__
 * 构造函数对象this指向obj,并实例化对象res
 * 确保实例化res是对象，可能是其他数据类型 null、undefined等等
 */

function New() {
  // 创建你一个空对象
  let obj  = new Object()

  // 获取构造函数
  let constructor = [].shift.call(arguments)
  // let constructor = Array.prototype.slice.call(arguments, 1)

  // 继承构造函数的原型
  obj.__proto__ = constructor.prototype
  // 绑定this, 创建实例
  const res = constructor.apply(obj, arguments)
  // 确保返回的是obj对象
  return res instanceof Object ? res  : obj 
}
