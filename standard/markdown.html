# 配置

module.exports = {
  title: 'Hello VuePress',
  description: 'Just playing around',
  themeConfig: {
    logo: '/assets/gitlab.png',
    nav: [
      { text: '首页', link: '/' },
      { text: '配置', link: '/config/' },
      { text: '恒生电子官网', link: 'http://www.hundsun.com/' },
      {
        text: '下拉',
        items: [
          { text: '首页', link: '/' },
          { text: '配置', link: '/config/' },
        ],
      },
    ],
    sidebar: {
      '/config/': [
        {
          title: '配置', // 必要的
          collapsable: false, // 可选的, 默认值是 true,
          children: ['', 'origin'],
        },
      ],
    },
  },
  markdown: {
    lineNumbers: true,
    anchor: { permalink: false },
    toc: { includeLevel: [1, 2, 3] },
    extendMarkdown: (md) => {
    },
  },
  configureWebpack: {
    resolve: {
      alias: {
        '@assets': '/assets',
      },
    },
  },
}
